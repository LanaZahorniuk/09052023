Создать файл script.sh где угодно.
Файл скрипта выполнит:
Создаст файл output.txt в домашней директории
Допишет в него информацию из вывода команды df -h, но только 4 столбец и вторую снизу строку
Заменит в полученном выводе букву М на kb и так же допишет результат в файл 

#!/bin/bash

# Создание файла output.txt в домашней директории
output_file="$HOME/output.txt"

# Получение информации из команды df -h, выбор 4-го столбца и второй строки снизу,
# замена буквы "М" на "kb"
df_output=$(df -h | awk '{print $4}' | sed -n '2p' | sed 's/М/kb/g')

# Добавление полученной информации в файл output.txt
echo "$df_output" >> "$output_file"
